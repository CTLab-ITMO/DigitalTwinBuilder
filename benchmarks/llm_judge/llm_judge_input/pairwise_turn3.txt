def simulate_sensors(t, dt):
    noise_temp = np.random.normal(0, 0.8)
    noise_speed = np.random.normal(0, 0.005)
    noise_water = np.random.normal(0, 8.0)
    noise_vib = np.random.normal(0, 0.06)

    sensor_data["temperature"] = TEMP_CRITICAL + noise_temp
    sensor_data["casting_speed"] = CASTING_SPEED_EMERGENCY + noise_speed
    sensor_data["water_flow"] = WATER_FLOW_CRITICAL + noise_water
    sensor_data["vibration"] = VIBRATION_CRITICAL + abs(noise_vib)
    sensor_data["timestamp"] = t

def check_critical():
    critical_flags = []

    temp = sensor_data["temperature"]
    if temp < TEMP_CRITICAL_THRESHOLD:
        msg = f"[CRITICAL] Temperature {temp:.1f}C < {TEMP_CRITICAL_THRESHOLD}C — EMERGENCY STOP REQUIRED"
        critical_flags.append(msg)
        warnings.warn(msg, RuntimeWarning)

    water = sensor_data["water_flow"]
    if water < WATER_FLOW_CRITICAL_THRESHOLD:
        msg = f"[CRITICAL] Water flow {water:.1f} m3/h < {WATER_FLOW_CRITICAL_THRESHOLD} m3/h — EMERGENCY STOP REQUIRED"
        critical_flags.append(msg)
        warnings.warn(msg, RuntimeWarning)

    vib = sensor_data["vibration"]
    if vib > VIBRATION_CRITICAL_THRESHOLD:
        msg = f"[CRITICAL] Vibration {vib:.2f} mm/s > {VIBRATION_CRITICAL_THRESHOLD} mm/s — EMERGENCY STOP REQUIRED"
        critical_flags.append(msg)
        warnings.warn(msg, RuntimeWarning)

    sensor_data["critical_active"] = len(critical_flags) > 0
    if sensor_data["critical_active"]:
        sensor_data["emergency_stop"] = True
    return critical_flags

def check_alarms():
    alarms = []
    temp = sensor_data["temperature"]
    if abs(temp - TEMP_NORMAL) > TEMP_ALARM_DELTA:
        msg = f"[ALARM] Temperature {temp:.1f}C — deviation {abs(temp - TEMP_NORMAL):.1f}C"
        alarms.append(msg)

    water = sensor_data["water_flow"]
    water_dev = abs(water - WATER_FLOW_NORMAL) / WATER_FLOW_NORMAL
    if water_dev > WATER_ALARM_PCT:
        msg = f"[ALARM] Water flow {water:.1f} m3/h — deviation {water_dev*100:.1f}%"
        alarms.append(msg)

    vib = sensor_data["vibration"]
    vib_factor = vib / VIBRATION_NORMAL
    if vib_factor >= VIBRATION_ALARM_FACTOR_LOW:
        msg = f"[ALARM] Vibration {vib:.2f} mm/s — factor {vib_factor:.2f}"
        alarms.append(msg)

    speed = sensor_data["casting_speed"]
    if speed <= CASTING_SPEED_EMERGENCY:
        msg = f"[ALARM] Casting speed {speed:.3f} m/min <= {CASTING_SPEED_EMERGENCY} m/min"
        alarms.append(msg)

    sensor_data["alarm_active"] = len(alarms) > 0
    return alarms

def predict_defects():
    temp_dev = abs(sensor_data["temperature"] - TEMP_NORMAL) / TEMP_ALARM_DELTA
    speed_norm = (sensor_data["casting_speed"] - CASTING_SPEED_MIN) / (CASTING_SPEED_MAX - CASTING_SPEED_MIN)
    speed_norm = np.clip(speed_norm, 0, 1)
    vib_factor = sensor_data["vibration"] / VIBRATION_NORMAL
    risk = temp_dev * 0.4 + (1.0 - speed_norm) * 0.3 + (vib_factor - 1.0) * 0.3
    sensor_data["defect_risk"] = np.clip(risk, 0.0, 1.0)
    return sensor_data["defect_risk"]

def update_wear(dt):
    vib_factor = sensor_data["vibration"] / VIBRATION_NORMAL
    wear_rate = 1e-6 * vib_factor * dt * 2.0
    sensor_data["equipment_wear"] += wear_rate

def update_chem_composition(t):
