def _utcnow():
    return datetime.now(timezone.utc)


def _get_const_function(value: float):
    # PyChrono API differs by version
    if hasattr(chrono, "ChFunction_Const"):
        return chrono.ChFunction_Const(value)
    if hasattr(chrono, "ChFunctionConst"):
        f = chrono.ChFunctionConst()
        f.SetConstant(value)
        return f
    raise RuntimeError("Cannot find a constant function class in this PyChrono build.")

class DatabaseClient:
    """
    Minimal DB client aligned to the provided schema.
    Expects tables to exist. Can optionally ensure presence (best-effort).
    """
    def __init__(self, dsn: str, plant_name: str, plant_location: str, plant_version: int, ensure_schema: bool = False):
        if psycopg2 is None:
            raise RuntimeError("psycopg2 is required for database logging but is not installed.")
        self.dsn = dsn
        self.plant_name = plant_name
        self.plant_location = plant_location
        self.plant_version = plant_version
        self.ensure_schema = ensure_schema

        self.conn = None
        self.cur = None
        self.plant_id = None
        self.sensor_uid_to_pk: Dict[int, int] = {}
        self._pending = queue.Queue(maxsize=100000)

    def connect(self):
        self.conn = psycopg2.connect(self.dsn)
        self.conn.autocommit = False
        self.cur = self.conn.cursor()
        if self.ensure_schema:
            self._ensure_schema_best_effort()
        self.plant_id = self._upsert_plant_metadata()
        self.conn.commit()

    def close(self):
        try:
            self.flush()
        except Exception:
            pass
        try:
            if self.cur:
                self.cur.close()
        finally:
            self.cur = None
        try:
            if self.conn:
                self.conn.close()
        finally:
            self.conn = None

    def _ensure_schema_best_effort(self):
        # Best-effort only; respects given schema but won't fight existing migrations.
        stmts = [
            """
            CREATE TABLE IF NOT EXISTS metadata (
                id SERIAL PRIMARY KEY,
                plant_name VARCHAR(255) NOT NULL,
                location VARCHAR(255) NOT NULL,
                version integer NOT NULL,
                last_updated TIMESTAMPTZ NOT NULL
            );
            """,
            """
            CREATE TABLE IF NOT EXISTS sensor_metadata (
                id SERIAL PRIMARY KEY,
                sensor_type VARCHAR(255) NOT NULL,
                location VARCHAR(255) NOT NULL,
                calibration_status BOOLEAN NOT NULL,
                sensor_id integer NOT NULL UNIQUE
            );
            """,
            """
            CREATE TABLE IF NOT EXISTS plant_data (
                id SERIAL PRIMARY KEY,
                timestamp TIMESTAMPTZ NOT NULL,
                value DOUBLE PRECISION NOT NULL,
                sensor_id integer NOT NULL REFERENCES sensor_metadata(id) ON DELETE CASCADE,
                plant_id integer NOT NULL REFERENCES metadata(id) ON DELETE CASCADE
            );
            """,
        ]
        for s in stmts:
            try:
                self.cur.execute(s)
                self.conn.commit()
            except Exception:
                self.conn.rollback()

    def _upsert_plant_metadata(self) -> int:
        now = _utcnow()
        # No uniqueness constraint specified in schema. We'll insert a row per run unless one matches name+location+version.
        # Use a select-first approach.
        self.cur.execute(
            """
            SELECT id FROM metadata
            WHERE plant_name=%s AND location=%s AND version=%s
            ORDER BY id DESC
            LIMIT 1;
            """,
            (self.plant_name, self.plant_location, self.plant_version),
        )
        row = self.cur.fetchone()
        if row:
            plant_id = int(row[0])
            self.cur.execute(
                "UPDATE metadata SET last_updated=%s WHERE id=%s;",
                (now, plant_id),
            )
            return plant_id

        self.cur.execute(
            """
            INSERT INTO metadata(plant_name, location, version, last_updated)
            VALUES (%s, %s, %s, %s)
            RETURNING id;
            """,
            (self.plant_name, self.plant_location, self.plant_version, now),
        )
        return int(self.cur.fetchone()[0])

    def ensure_sensor(self, s: SensorDef) -> int:
        """
        Ensures a row exists in sensor_metadata for the given sensor_uid (schema: sensor_id unique).
        Returns PK id (sensor_metadata.id), which is what plant_data.sensor_id references.
        """
        if s.sensor_uid in self.sensor_uid_to_pk:
            return self.sensor_uid_to_pk[s.sensor_uid]

        self.cur.execute(
            "SELECT id FROM sensor_metadata WHERE sensor_id=%s LIMIT 1;",
            (s.sensor_uid,),
        )
        row = self.cur.fetchone()
        if row:
            pk = int(row[0])
            # Keep metadata fresh
            self.cur.execute(
                """
                UPDATE sensor_metadata
                SET sensor_type=%s, location=%s, calibration_status=%s
                WHERE id=%s;
                """,
                (s.sensor_type, s.location, s.calibration_status, pk),
            )
            self.sensor_uid_to_pk[s.sensor_uid] = pk
            return pk

        self.cur.execute(
            """
            INSERT INTO sensor_metadata(sensor_type, location, calibration_status, sensor_id)
            VALUES (%s, %s, %s, %s)
            RETURNING id;
            """,
            (s.sensor_type, s.location, s.calibration_status, s.sensor_uid),
        )
        pk = int(self.cur.fetchone()[0])
        self.sensor_uid_to_pk[s.sensor_uid] = pk
        return pk

    def log_value(self, ts: datetime, value: float, sensor_pk: int):
        """
        Writes a plant_data row. On transient failure, queues data in memory.
        """
        item = (ts, float(value), int(sensor_pk), int(self.plant_id))
        try:
            self.cur.execute(
                """
                INSERT INTO plant_data(timestamp, value, sensor_id, plant_id)
                VALUES (%s, %s, %s, %s);
                """,
                item,
            )
        except Exception as e:
            self.conn.rollback()
            try:
                self._pending.put_nowait(item)
            except queue.Full:
                raise RuntimeError("DB logging failed and pending buffer is full.") from e

    def flush(self):
        if self.conn is None or self.cur is None:
            return
        # Flush pending first
        flushed = 0
        while True:
            try:
                item = self._pending.get_nowait()
            except queue.Empty:
                break
            self.cur.execute(
                """
                INSERT INTO plant_data(timestamp, value, sensor_id, plant_id)
                VALUES (%s, %s, %s, %s);
                """,
                item,
            )
            flushed += 1
        self.conn.commit()
        return flushed

    def _bounded(self, x: float, lo: float, hi: float) -> float:
        return min(max(x, lo), hi)

    def _n(self, sigma: float) -> float:
        # Approx normal via sum of uniforms (stable, avoids numpy dependency)
        s = 0.0
        for _ in range(6):
            s += (self.rng.random() - 0.5)
        return sigma * (s / 0.5) * 0.2

    def _update_surrogate_process(self, t_s: float, dt_s: float):
        """
        Lightweight coupled surrogate:
        - kiln temperature increases with fuel rate, decreases with feed rate and airflow mismatch
        - free lime decreases when temperature is adequate and residence/rotation is stable
        - CO increases when airflow is low relative to fuel (reducing atmosphere)
        - NOx increases with higher temperature and excess air
        - cyclone draft depends on airflow (more negative) + minor noise
        - mill load depends on feed - grinding capacity
        - fineness improves with higher cement mill speed and appropriate load
        """
        # Slowly vary "controls" to emulate operations (could be driven by real plant data)
        if t_s - self._last_control_update_s > 30.0:
            self._last_control_update_s = t_s
            self.fuel_rate = self._bounded(self.fuel_rate + self._n(0.02), 0.7, 1.3)
            self.raw_feed_rate = self._bounded(self.raw_feed_rate + self._n(0.02), 0.7, 1.3)
            self.airflow_rate = self._bounded(self.airflow_rate + self._n(0.02), 0.7, 1.3)

        # Kiln temperature dynamics
        target_temp = 1350.0 + 180.0 * self.fuel_rate - 80.0 * (self.raw_feed_rate - 1.0) + 40.0 * (self.airflow_rate - 1.0)
        target_temp = self._bounded(target_temp, 1200.0, 1550.0)
        tau = 120.0  # seconds
        self._kiln_temp_c += (target_temp - self._kiln_temp_c) * (dt_s / max(tau, 1e-6)) + self._n(0.3)

        # Free lime: higher when underburned (low temp) or unstable
        underburn = max(0.0, 1450.0 - self._kiln_temp_c) / 100.0
        overburn = max(0.0, self._kiln_temp_c - 1500.0) / 100.0
        self._free_lime_pct += (0.10 * underburn - 0.05 * (self._kiln_temp_c - 1450.0) / 100.0 + 0.02 * abs(self._n(0.1))) * (dt_s / 10.0)
        self._free_lime_pct += 0.08 * overburn * (dt_s / 10.0)  # overburn can also create poor reactivity / "apparent" lime issues
        self._free_lime_pct = self._bounded(self._free_lime_pct, 0.2, 3.5)

        # CO%: increases when fuel high and air low
        reducing_index = (self.fuel_rate / max(self.airflow_rate, 1e-3)) - 1.0
        co_target = 0.03 + 0.25 * max(0.0, reducing_index)
        self._co_pct += (co_target - self._co_pct) * (dt_s / 15.0) + self._n(0.002)
        self._co_pct = self._bounded(self._co_pct, 0.0, 1.0)

        # NOx ppm: increases with temperature and excess air
        excess_air = max(0.0, self.airflow_rate - self.fuel_rate)
        nox_target = 700.0 + 0.8 * (self._kiln_temp_c - 1350.0) + 500.0 * excess_air
        self._nox_ppm += (nox_target - self._nox_ppm) * (dt_s / 20.0) + self._n(3.0)
        self._nox_ppm = self._bounded(self._nox_ppm, 200.0, 2500.0)

        # Cyclone draft (Pa): negative, magnitude increases with airflow
        draft_target = -1800.0 - 1200.0 * (self.airflow_rate - 1.0)
        self._cyclone_draft_pa += (draft_target - self._cyclone_draft_pa) * (dt_s / 10.0) + self._n(15.0)
        self._cyclone_draft_pa = self._bounded(self._cyclone_draft_pa, -8000.0, -200.0)

        # Mill load (%): depends on feed and capacity (speed)
        capacity = 1.0 + 0.8 * (self.cement_mill_speed_rps / 0.22 - 1.0)
        load_target = 65.0 + 35.0 * (self.raw_feed_rate - capacity)
        self._mill_load_pct += (load_target - self._mill_load_pct) * (dt_s / 30.0) + self._n(0.2)
        self._mill_load_pct = self._bounded(self._mill_load_pct, 10.0, 95.0)

        # Fineness (Blaine, cm^2/g): higher fineness with higher speed and optimal load
        load_penalty = abs(self._mill_load_pct - 70.0) / 30.0
        fineness_target = 3200.0 + 400.0 * (self.cement_mill_speed_rps / 0.22 - 1.0) - 250.0 * load_penalty
        self._cement_fineness_blaine += (fineness_target - self._cement_fineness_blaine) * (dt_s / 120.0) + self._n(4.0)
        self._cement_fineness_blaine = self._bounded(self._cement_fineness_blaine, 2500.0, 4500.0)

    def step(self, dt_s: float):
        t0 = self.system.GetChTime()
        # Mechanical dynamics step
        self.system.DoStepDynamics(dt_s)
        t1 = self.system.GetChTime()
        # Surrogate process update using dt (in case integrator changes dt)
        self._update_surrogate_process(t1, float(t1 - t0))

    # --- Sensor readouts (virtual instrumentation) ---
    def read_kiln_temperature_c(self) -> float:
        return float(self._kiln_temp_c)

    def read_cyclone_draft_pa(self) -> float:
        return float(self._cyclone_draft_pa)

    def read_cement_fineness_blaine(self) -> float:
        return float(self._cement_fineness_blaine)

    def read_co_pct(self) -> float:
        return float(self._co_pct)

    def read_nox_ppm(self) -> float:
        return float(self._nox_ppm)

    def read_mill_load_pct(self) -> float:
        return float(self._mill_load_pct)

    # --- Critical parameter checks (soft alarms) ---
    def evaluate_alarms(self) -> List[str]:
        alarms = []
        if not (1450.0 <= self._kiln_temp_c <= 1480.0):
            alarms.append(f"Kiln sintering temperature out of band: {self._kiln_temp_c:.1f} °C (target 1450-1480)")
        if not (self._free_lime_pct < 1.5):
            alarms.append(f"Free lime high (proxy): {self._free_lime_pct:.2f}% (target < 1.5%)")
        if not (self._co_pct < 0.1):
            alarms.append(f"CO high: {self._co_pct:.3f}% (target < 0.1%)")
        return alarms


class SensorScheduler:
    def __init__(self, sensors: List[SensorDef]):
        self.sensors = sensors
        self.next_t: Dict[int, float] = {s.sensor_uid: 0.0 for s in sensors}

    def due(self, t_s: float) -> List[SensorDef]:
        out = []
        for s in self.sensors:
            nt = self.next_t[s.sensor_uid]
            if t_s + 1e-12 >= nt:
                out.append(s)
        return out

    def mark(self, sensor_uid: int, t_s: float, period_s: float):
        # Avoid drift: schedule next strictly in the future by period increments
        nt = self.next_t.get(sensor_uid, 0.0)
        if t_s + 1e-12 < nt:
            return
        if period_s <= 0:
            self.next_t[sensor_uid] = t_s
            return
        k = max(1, int(math.floor((t_s - nt) / period_s)) + 1)
        self.next_t[sensor_uid] = nt + k * period_s


def _build_sensors(twin: CementPlantDigitalTwin) -> List[SensorDef]:
    # Update frequencies per requirements:
    # - temperature in kiln: 1/min
    # - gas analysis: every 10 s
    # - lab: 1/hour (we map fineness to lab cadence; could also map free lime but it's not in sensor list)
    # Additional: cyclone draft every 10 s; mill load every 60 s for productivity control

def _configure_logging(verbosity: str):
    level = logging.INFO
    if verbosity.lower() == "debug":
        level = logging.DEBUG
    elif verbosity.lower() == "warning":
        level = logging.WARNING
    elif verbosity.lower() == "error":
        level = logging.ERROR

    logging.basicConfig(
        level=level,
        format="%(asctime)sZ %(levelname)s %(name)s: %(message)s",
        datefmt="%Y-%m-%dT%H:%M:%S",
    )
    # Force UTC-ish timestamps in logs
    logging.Formatter.converter = time.gmtime


def main():
    parser = argparse.ArgumentParser(description="PyChrono cement plant digital twin simulation with DB logging.")
    parser.add_argument("--duration", type=float, default=600.0, help="Simulation duration in seconds (default: 600).")
    parser.add_argument("--step", type=float, default=0.02, help="Simulation step size in seconds (default: 0.02).")
    parser.add_argument("--realtime", action="store_true", help="Sleep to approximate real-time execution.")
    parser.add_argument("--seed", type=int, default=42, help="Random seed for surrogate noise.")
    parser.add_argument("--verbosity", type=str, default="info", help="Logging level: debug|info|warning|error")
    parser.add_argument("--db-dsn", type=str, default=os.environ.get("DB_DSN", ""), help="PostgreSQL DSN string (or set DB_DSN).")
    parser.add_argument("--plant-name", type=str, default="Cement Plant Digital Twin", help="metadata.plant_name")
    parser.add_argument("--plant-location", type=str, default="RU", help="metadata.location")
    parser.add_argument("--plant-version", type=int, default=1, help="metadata.version")
    parser.add_argument("--ensure-schema", action="store_true", help="Best-effort create tables if missing.")
    parser.add_argument("--no-db", action="store_true", help="Run without database logging.")
    args = parser.parse_args()

    _configure_logging(args.verbosity)
    log = logging.getLogger("cement_twin")

    twin = CementPlantDigitalTwin(seed=args.seed)
    sensors = _build_sensors(twin)
    scheduler = SensorScheduler(sensors)

    db: Optional[DatabaseClient] = None
    sensor_pk: Dict[int, int] = {}

    shutting_down = {"flag": False}

    def _handle_signal(signum, frame):
        shutting_down["flag"] = True

    signal.signal(signal.SIGINT, _handle_signal)
    signal.signal(signal.SIGTERM, _handle_signal)

    sim_start_wall = _utcnow()
    last_commit_wall = time.time()
    commit_interval_s = 2.0

    try:
        if not args.no_db:
            if not args.db_dsn:
                raise RuntimeError("Database DSN is empty. Provide --db-dsn or set DB_DSN, or use --no-db.")
            db = DatabaseClient(
                dsn=args.db_dsn,
                plant_name=args.plant_name,
                plant_location=args.plant_location,
                plant_version=args.plant_version,
                ensure_schema=args.ensure_schema,
            )
            db.connect()
            for s in sensors:
                pk = db.ensure_sensor(s)
                sensor_pk[s.sensor_uid] = pk
            db.conn.commit()
            log.info("Database connected. plant_id=%s, sensors=%d", db.plant_id, len(sensors))
        else:
            log.warning("Running with --no-db (sensor data will not be persisted).")

        # Simulation loop
        target_dt = float(args.step)
        if target_dt <= 0:
            raise ValueError("--step must be > 0")

        end_t = float(args.duration)
        if end_t <= 0:
            raise ValueError("--duration must be > 0")

        last_rt = time.time()
        last_alarm_print_t = -1e9

        while True:
            if shutting_down["flag"]:
                log.warning("Shutdown requested by signal.")
                break

            t_s = float(twin.system.GetChTime())
            if t_s >= end_t:
                break

            # Step simulation
            twin.step(target_dt)

            t_s = float(twin.system.GetChTime())
            sim_ts = sim_start_wall + timedelta(seconds=t_s)

            # Read sensors that are due
            due = scheduler.due(t_s)
            if due:
                for s in due:
                    try:
                        val = float(s.value_fn())
                        if db is not None:
                            db.log_value(sim_ts, val, sensor_pk[s.sensor_uid])
                        scheduler.mark(s.sensor_uid, t_s, s.period_s)
                    except Exception as e:
                        log.exception("Sensor read/log failed (uid=%s, type=%s): %s", s.sensor_uid, s.sensor_type, e)
                        # keep scheduling so a single failure doesn't freeze that sensor
                        scheduler.mark(s.sensor_uid, t_s, s.period_s)

            # Commit periodically
            if db is not None:
                now = time.time()
                if now - last_commit_wall >= commit_interval_s:
                    try:
                        flushed = db.flush()
                        last_commit_wall = now
                        if args.verbosity.lower() == "debug" and flushed:
                            log.debug("DB flush committed. flushed=%d", flushed)
                    except Exception as e:
                        log.exception("DB flush failed: %s", e)
                        # keep running; data will queue (or raise if buffer full)

            # Soft alarms (do not spam)
            if t_s - last_alarm_print_t >= 10.0:
                alarms = twin.evaluate_alarms()
                if alarms:
                    for a in alarms:
                        log.warning("ALARM: %s", a)
                last_alarm_print_t = t_s

            # Real-time pacing
            if args.realtime:
                now = time.time()
                elapsed_rt = now - last_rt
                if elapsed_rt < target_dt:
                    time.sleep(max(0.0, target_dt - elapsed_rt))
                last_rt = time.time()

        log.info("Simulation finished at t=%.2fs (duration=%.2fs).", float(twin.system.GetChTime()), end_t)

    except KeyboardInterrupt:
        log.warning("Interrupted by user.")
    except Exception as e:
        log.exception("Fatal error: %s", e)
        raise
    finally:
        # Cleanup
        try:
            if db is not None:
                try:
                    db.flush()
                except Exception:
                    pass
                db.close()
                log.info("Database connection closed.")
        except Exception:
            pass


if __name__ == "__main__":
    main()
